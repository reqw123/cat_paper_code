<div class="stream-floating-panel" id="streamPanel">

    <div class="stream-title-bar" id="dragBar">
        <div class="stream-title" id="streamStatus">🔄 連線中...</div>

         ondblclick="toggleFullscreen()"
         onerror="streamOffline()"
         onload="streamOnline()">

<script>
(function(scope) {

    let panel = document.getElementById("streamPanel");
    let dragBar = document.getElementById("dragBar");

    let isDragging = false;
    let offsetX = 0;
    let offsetY = 0;

    window.streamOnline = function() {
        document.getElementById("streamImg").style.display = "block";
        document.getElementById("offlineMsg").style.display = "none";
        document.getElementById("streamStatus").innerHTML = "✅ 串流正常";
        document.getElementById("streamStatus").style.color = "#00ff88";
    };

    window.streamOffline = function() {
        document.getElementById("streamImg").style.display = "none";
        document.getElementById("offlineMsg").style.display = "block";
        document.getElementById("streamStatus").innerHTML = "❌ 串流離線";
        document.getElementById("streamStatus").style.color = "#ff5555";
    };

    window.toggleFullscreen = function() {
        if (!document.fullscreenElement) {
            panel.requestFullscreen();
        } else {
            document.exitFullscreen();
        }
    };

    window.toggleFloating = function() {
        panel.classList.toggle("floating");

        if (panel.classList.contains("floating")) {
            panel.style.top = "80px";
            panel.style.left = "80px";
            panel.style.width = "640px";
            panel.style.height = "420px";
        } else {
            panel.style.top = "";
            panel.style.left = "";
            panel.style.width = "100%";
            panel.style.height = "";
        }
    };

    window.reloadStream = function() {
        let img = document.getElementById("streamImg");
        let oldSrc = img.src;

        if (!oldSrc) return;

        img.src = "";
        setTimeout(function() {
            img.src = oldSrc.split("?")[0] + "?t=" + new Date().getTime();
        }, 300);

        document.getElementById("streamStatus").innerHTML = "🔄 重新連線中...";
        document.getElementById("streamStatus").style.color = "#ffaa00";
    };

    dragBar.addEventListener("mousedown", function(e) {
        if (!panel.classList.contains("floating")) return;

        isDragging = true;
        offsetX = e.clientX - panel.offsetLeft;
        offsetY = e.clientY - panel.offsetTop;

        e.preventDefault();
    });

    document.addEventListener("mousemove", function(e) {
        if (!isDragging) return;

        panel.style.left = (e.clientX - offsetX) + "px";
        panel.style.top = (e.clientY - offsetY) + "px";
    });

    document.addEventListener("mouseup", function() {
        isDragging = false;
    });

    scope.$watch("msg", function(msg) {
        if (!msg) return;

        let ip = msg.payload && msg.payload.ip ? msg.payload.ip : null;
        let streamUrl = msg.stream_url || null;

        let finalUrl = "";

        if (ip) {
            finalUrl = "http://" +" 192.168.0.120" + ":81/stream";
        } else if (streamUrl) {
            finalUrl = streamUrl;
        }

      
    });

})(scope);
</script>
